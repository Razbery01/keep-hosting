import { describe, it, expect, beforeAll, afterAll } from 'vitest'
import { createClient, SupabaseClient, User } from '@supabase/supabase-js'

const TEST_URL = process.env.TEST_SUPABASE_URL!
const TEST_ANON_KEY = process.env.TEST_SUPABASE_ANON_KEY!
const TEST_SERVICE_KEY = process.env.TEST_SUPABASE_SERVICE_ROLE_KEY!

const admin: SupabaseClient = createClient(TEST_URL, TEST_SERVICE_KEY, {
  auth: { autoRefreshToken: false, persistSession: false },
})

// Test user fixtures — recreated each run to avoid collision
const ts = Date.now()
const userAEmail = `rls-a-${ts}@example.test`
const userBEmail = `rls-b-${ts}@example.test`
const userPw = 'test-password-rls-12345!'

let userA: User
let userB: User
let orderAId: string
let orderBId: string
let siteAId: string
let siteBId: string
let subAId: string

async function signInAs(email: string, password: string): Promise<SupabaseClient> {
  const client = createClient(TEST_URL, TEST_ANON_KEY, {
    auth: { autoRefreshToken: false, persistSession: false },
  })
  const { error } = await client.auth.signInWithPassword({ email, password })
  if (error) throw new Error(`Sign-in failed for ${email}: ${error.message}`)
  return client
}

beforeAll(async () => {
  // Create two non-admin users
  const a = await admin.auth.admin.createUser({ email: userAEmail, password: userPw, email_confirm: true })
  if (a.error) throw a.error
  userA = a.data.user!

  const b = await admin.auth.admin.createUser({ email: userBEmail, password: userPw, email_confirm: true })
  if (b.error) throw b.error
  userB = b.data.user!

  // Ensure profiles exist with role='client' (migration may auto-create; upsert to be safe)
  await admin.from('profiles').upsert([
    { id: userA.id, email: userAEmail, role: 'client' },
    { id: userB.id, email: userBEmail, role: 'client' },
  ])

  // Seed an order per user
  const oA = await admin.from('orders')
    .insert({ user_id: userA.id, status: 'pending', plan: 'professional' })
    .select('id').single()
  if (oA.error) throw oA.error
  orderAId = oA.data.id

  const oB = await admin.from('orders')
    .insert({ user_id: userB.id, status: 'pending', plan: 'professional' })
    .select('id').single()
  if (oB.error) throw oB.error
  orderBId = oB.data.id

  // Seed a client_sites row per user
  const sA = await admin.from('client_sites')
    .insert({ user_id: userA.id, order_id: orderAId, business_name: 'A Co' })
    .select('id').single()
  if (sA.error) throw sA.error
  siteAId = sA.data.id

  const sB = await admin.from('client_sites')
    .insert({ user_id: userB.id, order_id: orderBId, business_name: 'B Co' })
    .select('id').single()
  if (sB.error) throw sB.error
  siteBId = sB.data.id

  // Seed a file_uploads row per user
  await admin.from('file_uploads').insert([
    { user_id: userA.id, site_id: siteAId, file_name: 'a.jpg', file_path: `${userA.id}/${siteAId}/logo-xxx.jpg` },
    { user_id: userB.id, site_id: siteBId, file_name: 'b.jpg', file_path: `${userB.id}/${siteBId}/logo-yyy.jpg` },
  ])

  // Seed a subscription per user (Plan 02 table)
  const subA = await admin.from('subscriptions')
    .insert({ user_id: userA.id, order_id: orderAId, status: 'active' })
    .select('id').single()
  if (subA.error) throw subA.error
  subAId = subA.data.id

  await admin.from('subscriptions')
    .insert({ user_id: userB.id, order_id: orderBId, status: 'active' })
}, 60000)

afterAll(async () => {
  // Deleting users cascades to orders/sites/uploads/subscriptions via ON DELETE CASCADE
  if (userA?.id) await admin.auth.admin.deleteUser(userA.id).catch(() => {})
  if (userB?.id) await admin.auth.admin.deleteUser(userB.id).catch(() => {})
}, 60000)

describe('SEC-05: Non-admin RLS isolation', () => {
  it('profiles: non-admin reads own, 0 rows for other user', async () => {
    const a = await signInAs(userAEmail, userPw)
    const own = await a.from('profiles').select('id').eq('id', userA.id)
    expect(own.data).toHaveLength(1)

    const other = await a.from('profiles').select('id').eq('id', userB.id)
    expect(other.data ?? []).toHaveLength(0)
  })

  it('orders: non-admin reads own, 0 rows for other user', async () => {
    const a = await signInAs(userAEmail, userPw)
    const own = await a.from('orders').select('id').eq('id', orderAId)
    expect(own.data).toHaveLength(1)

    const other = await a.from('orders').select('id').eq('id', orderBId)
    expect(other.data ?? []).toHaveLength(0)
  })

  it('client_sites: non-admin reads own, 0 rows for other user', async () => {
    const a = await signInAs(userAEmail, userPw)
    const own = await a.from('client_sites').select('id').eq('id', siteAId)
    expect(own.data).toHaveLength(1)

    const other = await a.from('client_sites').select('id').eq('id', siteBId)
    expect(other.data ?? []).toHaveLength(0)
  })

  it('file_uploads: non-admin reads own, 0 rows for other user', async () => {
    const a = await signInAs(userAEmail, userPw)
    const own = await a.from('file_uploads').select('id').eq('user_id', userA.id)
    expect((own.data ?? []).length).toBeGreaterThanOrEqual(1)

    const other = await a.from('file_uploads').select('id').eq('user_id', userB.id)
    expect(other.data ?? []).toHaveLength(0)
  })

  it('subscriptions: non-admin reads own, 0 rows for other user', async () => {
    const a = await signInAs(userAEmail, userPw)
    const own = await a.from('subscriptions').select('id').eq('user_id', userA.id)
    expect((own.data ?? []).length).toBeGreaterThanOrEqual(1)

    const other = await a.from('subscriptions').select('id').eq('user_id', userB.id)
    expect(other.data ?? []).toHaveLength(0)
  })
})

const TEST_ANON_KEY = process.env.TEST_SUPABASE_ANON_KEY!
const TEST_SERVICE_KEY = process.env.TEST_SUPABASE_SERVICE_ROLE_KEY!

beforeAll(async () => {

  adminUser = aResp.data.user!

  const cA = await admin.auth.admin.createUser({ email: clientAEmail, password: pw, email_confirm: true })
  if (cA.error) throw cA.error

  await admin.from('profiles').upsert([
    { id: adminUser.id, email: adminEmail, role: 'admin' },
    { id: clientAUser.id, email: clientAEmail, role: 'client' },
    { id: clientBUser.id, email: clientBEmail, role: 'client' },
  ])

  const oA = await admin.from('orders')
    .insert({ user_id: clientAUser.id, status: 'pending', plan: 'professional' })
    .select('id').single()
  if (oA.error) throw oA.error
  orderAId = oA.data.id

  const oB = await admin.from('orders')
    .insert({ user_id: clientBUser.id, status: 'paid', plan: 'professional' })
    .select('id').single()
  if (oB.error) throw oB.error
  orderBId = oB.data.id
}, 60000)

afterAll(async () => {
  if (adminUser?.id) await admin.auth.admin.deleteUser(adminUser.id).catch(() => {})
  if (clientAUser?.id) await admin.auth.admin.deleteUser(clientAUser.id).catch(() => {})
  if (clientBUser?.id) await admin.auth.admin.deleteUser(clientBUser.id).catch(() => {})

describe('SEC-02: Admin can read across users via is_admin() RLS helper', () => {

    const all = await a.from('profiles').select('id').in('id', [clientAUser.id, clientBUser.id])
    expect(all.data ?? []).toHaveLength(2)
  })

  it('admin sees orders from both client users', async () => {
    const a = await signInAs(adminEmail)
    const all = await a.from('orders').select('id').in('id', [orderAId, orderBId])
    expect(all.data ?? []).toHaveLength(2)
  })

  it('client user does NOT see admin profile via anon client', async () => {
    const c = await signInAs(clientAEmail)
    const r = await c.from('profiles').select('id').eq('id', adminUser.id)
    expect(r.data ?? []).toHaveLength(0)
  })
})
